FROM debian
RUN apt-get update && apt-get install -y 
   curl 
   gcc 
   libpcre3-dev 
   libpcre3 
   libssl-dev 
   zlib1g-dev 
   make
   python

RUN curl http://tengine.taobao.org/download/tengine-2.3.0.tar.gz > /opt/tengine$

WORKDIR /opt

RUN tar xzf tengine-2.3.0.tar.gz

WORKDIR /opt/tengine-2.3.0

RUN ./configure
